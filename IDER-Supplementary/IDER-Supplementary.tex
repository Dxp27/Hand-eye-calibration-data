\begin{algorithm}[!t]
	\caption{IDER Algorithm.}\label{alg:ider}
	\begin{algorithmic}
		\STATE \textbf{Input:} \(x_0, A, B\)
		\STATE {\textsc{Initialization:}}
		\STATE 1. \hspace{0.5cm} \textbf{Initialize:} \(F, CR, T, N_P\)
		\STATE 2. \hspace{0.5cm} \textbf{Initialize:} \(x_{de} = x_0\), \(f(X_{de}) = f(X_0)\)
		\STATE {\textsc{Optimization Step:}}
		\STATE 3. \hspace{0.5cm} \textbf{for} \(t = 1\) \textbf{to} \(T\) 
		\STATE 4. \hspace{1.0cm} Update \(x_i\) with Eq. (23)
		\STATE 5. \hspace{1.0cm} Calculate \(v_i\) with Eq. (26)
		\STATE 6. \hspace{1.0cm} Update \(u_i\) with Eq. (28)
		\STATE 7. \hspace{1.0cm} Update \(u_i\) with Eq. (30)
		\STATE 8. \hspace{1.0cm} Calculate \(f_{ui}\) with Eq. (25)
		\STATE 9. \hspace{1.0cm} Update \(x_{(t+1)}\) with Eq. (31)
		\STATE 10. \hspace{0.5cm} \textbf{end for}
		\STATE 11. \hspace{0.5cm} \(x_{de} = x_{(t+1)}\)
		\STATE {\textsc{Regularization Step:}}
		\STATE 12. \hspace{0.5cm} \textbf{for} \(t = 1\) \textbf{to} \(T\)
		\STATE 13. \hspace{1.0cm} \textbf{if} \(|E_1| \geq 1\)
		\STATE 14. \hspace{1.5cm} Update \(P_i^{(t+1)}\) with Eq. (38)
		\STATE 15. \hspace{1.0cm} \textbf{else}
		\STATE 16. \hspace{1.5cm} Update \(P_i^{(t+1)}\) with Eq. (39)
		\STATE 17. \hspace{1.0cm} \textbf{end if}
		\STATE 18. \hspace{1.0cm} Calculate \(F(\alpha_{\text{opt}}, \lambda_{\text{opt}})\) with Eq. (34)
		\STATE 19. \hspace{1.0cm} Update \((\alpha^*, \lambda^*)\) with Eq. (42)
		\STATE 20. \hspace{1.0cm} Calculate \(x_{en}\) with Eq. (33)
		\STATE 21. \hspace{1.0cm} Update \(X_{en}\)
		\STATE 22. \hspace{1.0cm} Calculate RMSE with Eq. (43)
		\STATE 23. \hspace{0.5cm} \textbf{end for}
		\STATE \textbf{Output:} \(X_{en}\), RMSE
	\end{algorithmic}
\end{algorithm}
